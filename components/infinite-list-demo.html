  </InfiniteListDemo>
</template>

<script setup>
import { computed } from 'vue'
const items = computed(() => {
  // 自己模拟一万条数据，将其内容进行填充

    .fill('')
    .map((item, index) => ({
      id: index,
      content: '列表项内容' + index,
    }))
})
</script>
